local function hover()
    local plr = game.Players.LocalPlayer
    local enabled = false
    local platform = nil
    local platformCon = nil
    local characterCon = nil

    local function createPlatform()
        if platform then platform:Destroy() end
        platform = Instance.new("Part")
        platform.Size = Vector3.new(3, 0.2, 3)
        platform.Transparency = 1
        platform.Anchored = true
        platform.CanCollide = true
        platform.Parent = workspace
    end

    local function updatePlatform()
        local char = plr.Character
        local root = char and char:FindFirstChild("HumanoidRootPart")
        if root and platform then
            local pos = root.Position
            local vel = root.AssemblyLinearVelocity
            if vel.Y > 0 then
                -- 跳跃时让平台跟随上升，但保持紧贴脚底
                platform.CFrame = CFrame.new(pos.X, pos.Y - 1.5, pos.Z)
            else
                platform.CFrame = CFrame.new(pos.X, pos.Y - 1.5, pos.Z)
            end
        end
    end

    local function start()
        if enabled then return end
        enabled = true

        if not plr.Character then
            plr.CharacterAdded:Wait()
        end

        createPlatform()
        updatePlatform()

        platformCon = game:GetService("RunService").Heartbeat:Connect(updatePlatform)

        characterCon = plr.CharacterAdded:Connect(function()
            createPlatform()
            updatePlatform()
        end)
    end

    local function stop()
        enabled = false
        if platformCon then platformCon:Disconnect() platformCon = nil end
        if characterCon then characterCon:Disconnect() characterCon = nil end
        if platform then platform:Destroy() platform = nil end
    end

    return {start = start, stop = stop}
end

local api = hover()
if api then getgenv().HoverAPI = api end

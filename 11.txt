local function playerControl()
    local plr = game.Players.LocalPlayer
    local selectedPlayer = nil

    local function getPlayerList()
        local list = {}
        for _, p in pairs(game.Players:GetPlayers()) do
            if p ~= plr then
                table.insert(list, p.Name)
            end
        end
        return list
    end

    local function selectPlayer(name)
        local target = game.Players:FindFirstChild(name)
        if target then
            selectedPlayer = target
            return true, "已选择 " .. target.Name
        end
        return false, "玩家不存在"
    end

    local function getSelected()
        return selectedPlayer
    end

    local function teleportTo(target)
        target = target or selectedPlayer
        if not target then return false, "未选择玩家" end
        local char = target.Character
        local root = char and char:FindFirstChild("HumanoidRootPart")
        if not root then return false, "玩家没有角色" end
        local myChar = plr.Character
        local myRoot = myChar and myChar:FindFirstChild("HumanoidRootPart")
        if not myRoot then return false, "你没有角色" end
        myRoot.CFrame = root.CFrame + Vector3.new(0, 2, 0)
        return true, "已传送到 " .. target.Name
    end

    return {
        getPlayerList = getPlayerList,
        selectPlayer = selectPlayer,
        getSelected = getSelected,
        teleportTo = teleportTo,
    }
end

local api = playerControl()
if api then getgenv().PlayerControl = api end

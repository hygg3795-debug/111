local function ringControl()
    local plr = game.Players.LocalPlayer
    local enabled = false
    local parts = {}
    local config = {radius = 50, height = 100, rotationSpeed = 10, attractionStrength = 1000}
    local con = nil
    local partCon = nil
    local removeCon = nil
    local screenGui = nil
    local mainFrame = nil

    local function retainPart(part)
        if part:IsA("BasePart") and not part.Anchored and part:IsDescendantOf(workspace) then
            if part.Parent == plr.Character or part:IsDescendantOf(plr.Character) then return false end
            pcall(function()
                part.CustomPhysicalProperties = PhysicalProperties.new(0, 0, 0, 0, 0)
                part.CanCollide = false
            end)
            return true
        end
        return false
    end

    local function addPart(part)
        if retainPart(part) and not table.find(parts, part) then
            table.insert(parts, part)
        end
    end

    local function removePart(part)
        local idx = table.find(parts, part)
        if idx then table.remove(parts, idx) end
    end

    for _, part in pairs(workspace:GetDescendants()) do addPart(part) end
    if partCon then partCon:Disconnect() end
    partCon = workspace.DescendantAdded:Connect(addPart)
    if removeCon then removeCon:Disconnect() end
    removeCon = workspace.DescendantRemoving:Connect(removePart)

    local function createGUI()
        if screenGui then return end
        screenGui = Instance.new("ScreenGui")
        screenGui.Name = "SuperRingPartsGUI"
        screenGui.ResetOnSpawn = false
        screenGui.Parent = plr:WaitForChild("PlayerGui")

        mainFrame = Instance.new("Frame")
        mainFrame.Size = UDim2.new(0, 300, 0, 500)
        mainFrame.Position = UDim2.new(0.5, -150, 0.5, -250)
        mainFrame.BorderSizePixel = 0
        mainFrame.Parent = screenGui
        mainFrame.Active = true
        mainFrame.Draggable = true

        local uiCorner = Instance.new("UICorner")
        uiCorner.CornerRadius = UDim.new(0, 20)
        uiCorner.Parent = mainFrame

        local title = Instance.new("TextLabel")
        title.Size = UDim2.new(1, 0, 0, 40)
        title.Text = "🔴 黑洞1 控制面板 🔴"
        title.TextColor3 = Color3.fromRGB(0, 0, 0)
        title.BackgroundColor3 = Color3.fromRGB(0, 204, 204)
        title.Font = Enum.Font.Fondamento
        title.TextSize = 22
        title.Parent = mainFrame
        local titleCorner = Instance.new("UICorner")
        titleCorner.CornerRadius = UDim.new(0, 20)
        titleCorner.Parent = title

        local closeBtn = Instance.new("TextButton")
        closeBtn.Size = UDim2.new(0, 30, 0, 30)
        closeBtn.Position = UDim2.new(1, -35, 0, 5)
        closeBtn.Text = "✕"
        closeBtn.TextColor3 = Color3.fromRGB(255, 255, 255)
        closeBtn.BackgroundColor3 = Color3.fromRGB(200, 50, 50)
        closeBtn.Font = Enum.Font.Fondamento
        closeBtn.TextSize = 15
        closeBtn.Parent = mainFrame
        local closeCorner = Instance.new("UICorner")
        closeCorner.CornerRadius = UDim.new(0, 15)
        closeCorner.Parent = closeBtn
        closeBtn.MouseButton1Click:Connect(function()
            if screenGui then screenGui:Destroy() screenGui = nil end
        end)

        local function createControl(labelText, configKey, posY, color, step, minVal, maxVal)
            step = step or 10
            minVal = minVal or 0
            maxVal = maxVal or 10000

            local bg = Instance.new("Frame")
            bg.Size = UDim2.new(0.9, 0, 0, 42)
            bg.Position = UDim2.new(0.05, 0, posY, 0)
            bg.BackgroundColor3 = color
            bg.BorderSizePixel = 0
            bg.Parent = mainFrame
            local bgCorner = Instance.new("UICorner")
            bgCorner.CornerRadius = UDim.new(0, 8)
            bgCorner.Parent = bg

            local label = Instance.new("TextLabel")
            label.Size = UDim2.new(0.45, 0, 1, 0)
            label.Position = UDim2.new(0.02, 0, 0, 0)
            label.Text = labelText
            label.TextColor3 = Color3.fromRGB(0, 0, 0)
            label.BackgroundTransparency = 1
            label.Font = Enum.Font.Fondamento
            label.TextSize = 18
            label.TextXAlignment = Enum.TextXAlignment.Left
            label.Parent = bg

            local display = Instance.new("TextLabel")
            display.Size = UDim2.new(0.2, 0, 1, 0)
            display.Position = UDim2.new(0.42, 0, 0, 0)
            display.Text = tostring(config[configKey])
            display.TextColor3 = Color3.fromRGB(0, 0, 0)
            display.BackgroundTransparency = 1
            display.Font = Enum.Font.Fondamento
            display.TextSize = 18
            display.Parent = bg

            local minusBtn = Instance.new("TextButton")
            minusBtn.Size = UDim2.new(0, 30, 0, 30)
            minusBtn.Position = UDim2.new(0.7, 0, 0.5, -15)
            minusBtn.Text = "-"
            minusBtn.TextColor3 = Color3.fromRGB(0, 0, 0)
            minusBtn.BackgroundColor3 = Color3.fromRGB(200, 200, 200)
            minusBtn.Font = Enum.Font.Fondamento
            minusBtn.TextSize = 20
            minusBtn.Parent = bg
            local mc = Instance.new("UICorner")
            mc.CornerRadius = UDim.new(0, 8)
            mc.Parent = minusBtn

            local plusBtn = Instance.new("TextButton")
            plusBtn.Size = UDim2.new(0, 30, 0, 30)
            plusBtn.Position = UDim2.new(0.82, 0, 0.5, -15)
            plusBtn.Text = "+"
            plusBtn.TextColor3 = Color3.fromRGB(0, 0, 0)
            plusBtn.BackgroundColor3 = Color3.fromRGB(200, 200, 200)
            plusBtn.Font = Enum.Font.Fondamento
            plusBtn.TextSize = 20
            plusBtn.Parent = bg
            local pc = Instance.new("UICorner")
            pc.CornerRadius = UDim.new(0, 8)
            pc.Parent = plusBtn

            local inputBox = Instance.new("TextBox")
            inputBox.Size = UDim2.new(0.2, 0, 0, 30)
            inputBox.Position = UDim2.new(0.42, 0, 0.5, -15)
            inputBox.Text = tostring(config[configKey])
            inputBox.TextColor3 = Color3.fromRGB(0, 0, 0)
            inputBox.BackgroundColor3 = Color3.fromRGB(255, 255, 255)
            inputBox.Font = Enum.Font.Fondamento
            inputBox.TextSize = 16
            inputBox.TextXAlignment = Enum.TextXAlignment.Center
            inputBox.PlaceholderText = "数值"
            inputBox.ClearTextOnFocus = false
            inputBox.Parent = bg
            local ibCorner = Instance.new("UICorner")
            ibCorner.CornerRadius = UDim.new(0, 6)
            ibCorner.Parent = inputBox

            local function updateDisplay()
                display.Text = tostring(config[configKey])
                inputBox.Text = tostring(config[configKey])
                local method = "set" .. configKey:gsub("^%l", string.upper)
                if enabled and _G.RingControl and _G.RingControl[method] then
                    _G.RingControl[method](config[configKey])
                end
            end

            minusBtn.MouseButton1Click:Connect(function()
                local newVal = math.max(minVal, config[configKey] - step)
                config[configKey] = newVal
                updateDisplay()
            end)

            plusBtn.MouseButton1Click:Connect(function()
                local newVal = math.min(maxVal, config[configKey] + step)
                config[configKey] = newVal
                updateDisplay()
            end)

            inputBox.FocusLost:Connect(function(enterPressed)
                if enterPressed then
                    local newVal = tonumber(inputBox.Text)
                    if newVal then
                        newVal = math.clamp(newVal, minVal, maxVal)
                        config[configKey] = newVal
                        updateDisplay()
                    else
                        inputBox.Text = tostring(config[configKey])
                    end
                end
            end)
        end

        createControl("半径", "radius", 0.08, Color3.fromRGB(153, 153, 0), 10, 0, 1000)
        createControl("高度", "height", 0.22, Color3.fromRGB(153, 0, 153), 10, 0, 1000)
        createControl("转速", "rotationSpeed", 0.36, Color3.fromRGB(0, 153, 153), 1, 0, 200)
        createControl("吸力", "attractionStrength", 0.50, Color3.fromRGB(153, 0, 0), 100, 0, 50000)

        local hue = 0
        game:GetService("RunService").Heartbeat:Connect(function()
            if not mainFrame or not mainFrame.Parent then return end
            hue = (hue + 0.01) % 1
            mainFrame.BackgroundColor3 = Color3.fromHSV(hue, 1, 1)
        end)

        local textHue = 0
        game:GetService("RunService").Heartbeat:Connect(function()
            if not title or not title.Parent then return end
            textHue = (textHue + 0.01) % 1
            title.TextColor3 = Color3.fromHSV(textHue, 1, 1)
        end)
    end

    local function start()
        if con then return end
        enabled = true
        createGUI()
        con = game:GetService("RunService").Heartbeat:Connect(function()
            if not enabled then return end
            local root = plr.Character and plr.Character:FindFirstChild("HumanoidRootPart")
            if not root then return end
            local center = root.Position
            for _, part in pairs(parts) do
                if part.Parent and not part.Anchored then
                    local pos = part.Position
                    local dist = (Vector3.new(pos.X, center.Y, pos.Z) - center).Magnitude
                    local angle = math.atan2(pos.Z - center.Z, pos.X - center.X)
                    local newAngle = angle + math.rad(config.rotationSpeed)
                    local targetPos = Vector3.new(
                        center.X + math.cos(newAngle) * math.min(config.radius, dist),
                        center.Y + (config.height * (math.abs(math.sin((pos.Y - center.Y) / config.height)))),
                        center.Z + math.sin(newAngle) * math.min(config.radius, dist)
                    )
                    local dir = (targetPos - part.Position).unit
                    part.Velocity = dir * config.attractionStrength
                end
            end
        end)
    end

    local function stop()
        enabled = false
        if con then con:Disconnect() con = nil end
        if screenGui then screenGui:Destroy() screenGui = nil end
        mainFrame = nil
    end

    return {
        start = start,
        stop = stop,
        setRadius = function(v) config.radius = v end,
        setHeight = function(v) config.height = v end,
        setSpeed = function(v) config.rotationSpeed = v end,
        setStrength = function(v) config.attractionStrength = v end,
    }
end

local api = ringControl()
if api then getgenv().RingControl = api end

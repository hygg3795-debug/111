local function ringControl()local plr=game.Players.LocalPlayer local enabled=false local parts={} local config={radius=50,height=100,rotationSpeed=10,attractionStrength=1000}local con=nil local partCon=nil local removeCon=nil local function retainPart(part)if part:IsA("BasePart")and not part.Anchored and part:IsDescendantOf(workspace)then if part.Parent==plr.Character or part:IsDescendantOf(plr.Character)then return false end pcall(function()part.CustomPhysicalProperties=PhysicalProperties.new(0,0,0,0,0)part.CanCollide=false end)return true end return false end local function addPart(part)if retainPart(part)and not table.find(parts,part)then table.insert(parts,part)end end local function removePart(part)local idx=table.find(parts,part)if idx then table.remove(parts,idx)end end for _,part in pairs(workspace:GetDescendants())do addPart(part)end if partCon then partCon:Disconnect()end partCon=workspace.DescendantAdded:Connect(addPart)if removeCon then removeCon:Disconnect()end removeCon=workspace.DescendantRemoving:Connect(removePart)local function start()if con then return end enabled=true con=game:GetService("RunService").Heartbeat:Connect(function()if not enabled then return end local root=plr.Character and plr.Character:FindFirstChild("HumanoidRootPart")if not root then return end local center=root.Position for _,part in pairs(parts)do if part.Parent and not part.Anchored then local pos=part.Position local dist=(Vector3.new(pos.X,center.Y,pos.Z)-center).Magnitude local angle=math.atan2(pos.Z-center.Z,pos.X-center.X)local newAngle=angle+math.rad(config.rotationSpeed)local targetPos=Vector3.new(center.X+math.cos(newAngle)*math.min(config.radius,dist),center.Y+(config.height*(math.abs(math.sin((pos.Y-center.Y)/config.height)))),center.Z+math.sin(newAngle)*math.min(config.radius,dist))local dir=(targetPos-part.Position).unit part.Velocity=dir*config.attractionStrength end end end)end local function stop()enabled=false if con then con:Disconnect()con=nil end end return{start=start,stop=stop,setRadius=function(v)config.radius=v end,setHeight=function(v)config.height=v end,setSpeed=function(v)config.rotationSpeed=v end,setStrength=function(v)config.attractionStrength=v end}end local api=ringControl()if api then getgenv().RingControl=api end
